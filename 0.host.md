```bash
apt install -y sshpass

PASSWORD="root"

HOSTS_CONTENT=$(cat <<EOF
192.168.29.61 etcdone
192.168.29.62 etcdtwo
192.168.29.63 etcdthree
192.168.29.51 masterone
192.168.29.71 workerone
EOF
)

for node in 192.168.29.51 192.168.29.61 192.168.29.62 192.168.29.63 192.168.29.71; do
  sshpass -p "$PASSWORD" ssh -o StrictHostKeyChecking=no root@$node "echo \"$HOSTS_CONTENT\" >> /etc/hosts"
done
```
